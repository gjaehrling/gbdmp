library("dplyr")

pf.fc_by_age_gender <- pf %.% filter(!is.na(gender)) %.% group_by(age, gender) %.% 
    summarize(mean_friend_count = mean(friend_count), median_friend_count = median(friend_count), 
        n = n()) %.% ungroup() %.% arrange(age)
